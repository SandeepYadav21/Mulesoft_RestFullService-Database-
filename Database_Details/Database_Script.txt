#Stored Procedure creation to delet the employee record:
DELIMITER //
CREATE PROCEDURE delete_emp (
    IN p_id INT,
    IN p_name VARCHAR(50)
)
BEGIN
    DELETE FROM emp
    WHERE emp_id = p_id
      AND emp_name = p_name;
END //
DELIMITER ;

CALL delete_emp(100, 'Sandeep');
